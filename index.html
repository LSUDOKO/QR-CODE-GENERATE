Code Examples


import qrcode
from PIL import Image

qr = qrcode.QRCode(
    version=1,
    error_correction=qrcode.constants.ERROR_CORRECT_L,
    box_size=10,
    border=4,
)
qr.add_data('https://example.com')
qr.make(fit=True)

img = qr.make_image(fill_color="black", back_color="white")
img.save("basic_qrcode.png")
                        

import segno

qrcode = segno.make('https://example.com')
qrcode.save(
    'colored_qrcode.png',
    scale=10,
    light='lightblue',
    dark='darkblue'
)
                        

import segno
from pathlib import Path

qrcode = segno.make('https://example.com')
out = Path('animated_qrcode.gif')

# Create animated QR code (GIF)
qrcode.to_artistic(
    background='images/background.png',
    target=out,
    scale=10,
    light='white',
    dark='black'
)
